Library Max

Require Import Omega.
Require Import Max.

Lemma max_upper:
 ∀ a b n, a ≤ n → b ≤ n → max a b ≤ n.

Lemma max_upbound_l x y z:
 max x y ≤ z →
  x ≤ z.

Lemma max_upbound_r x y z:
 max x y ≤ z →
  y ≤ z.

Hint Resolve max_upbound_l max_upbound_r max_upper.
