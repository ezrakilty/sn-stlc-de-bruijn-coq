Library TakeDrop

Require Import List.

Drop the first k elements of a list.
Fixpoint drop A k (xs:list A) {struct k} :=
match k with
| 0 ⇒ xs
| S k' ⇒ match xs with
    | x::xs ⇒ drop A k' xs
    | nil ⇒ nil
    end
end.

Take just the first k elements of a list (dropping the rest).
Fixpoint take A k (xs:list A) {struct k} :=
match k with
| 0 ⇒ nil
| S k' ⇒ match xs with
    | x::xs ⇒ x :: take A k' xs
    | nil ⇒ nil
    end
end.

Lemma take_drop :
  ∀ A n xs, take A n xs ++ drop A n xs = xs.

Lemma length_take :
  ∀ A k (xs : list A), length (take A k xs) = k.

Hint Rewrite take_drop length_take : lists.
