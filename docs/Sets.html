Fixpoint set_remove A (Aeq_dec:∀ (x y: A), {x=y}+{x≠y}) (a:A) (x:set A)
  : set A :=
  match x with
  | nil ⇒ empty_set A
  | a1 :: x1 ⇒
    match Aeq_dec a a1 with
    | left _ ⇒ set_remove _ Aeq_dec a x1
    | right _ ⇒ a1 :: set_remove _ Aeq_dec a x1
    end
  end.
