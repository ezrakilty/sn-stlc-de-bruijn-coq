Library RangeSet

Require Import Coq.Arith.Arith.
Require Import Coq.Lists.List.

Fixpoint range_set min max :=
  match max with
    | S max_dec ⇒
      if le_gt_dec max min then nil : list nat else range_set min max_dec
    | Z ⇒ nil
  end.

Lemma range_set_n_n_empty:
  ∀ n,
    range_set n n = nil.

Lemma range_set_singleton:
  ∀ n,
    range_set n (S n) = n::nil.
