Lemma map_lemma:
  ∀ (A B :Set) (f:A→B) (x:A) xs,
    map f (x::xs) = (f x) :: (map f xs).

Lemma map_app_commutes:
  ∀ (A B:Set) f l1 l2,
    map (A:=A)(B:=B) f (l1 ++ l2) = map f l1 ++ map f l2.

Lemma map_idy : ∀ A xs, map (idy A) xs = xs.

Lemma map_extensionality:
  ∀ A B (f g:A→B) xs,
    (∀ x, set_In x xs → f x = g x)
     → map f xs = map g xs.

Lemma set_map_extensionality:
  ∀ A B B_eq_dec (f g:A→B) xs,
    (∀ x, set_In x xs → f x = g x)
     → set_map B_eq_dec f xs = set_map B_eq_dec g xs.

Lemma map_image:
  ∀ A B (f:A→B) x xs,
    set_In x (map f xs) → ∃ x', x = f x' ∧ set_In x' xs.

Lemma set_map_idy_ext :
  ∀ A eq_dec (f:A→A) xs,
    
    (∀ x, true = set_mem eq_dec x xs → f x = x)
    → eq_sets _ (set_map eq_dec f xs) xs.

Lemma set_map_idy :
  ∀ A eq_dec (xs:set A),
    eq_sets _ (set_map eq_dec (idy A) xs) xs.

Lemma set_map_map :
  ∀ A B eq_dec f (g:A→B) xs,
    eq_sets _
      (set_map eq_dec f (set_map eq_dec g xs))
      (set_map eq_dec (fun x ⇒ f (g x)) xs).
